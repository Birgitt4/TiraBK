45      public void shuffle() {
46          game.suffle();
47          while (!game.isSolvable()) {
48              game.suffle();
49          }
50          //Järkevä tapa alustaa sama järjestys tilesseille?
51          
52          for (int i = 0; i < 16; i++) {
53              if (game.getGrid()[i] == 0) {
54                  continue;
55              }
56              
57              int tileNumber = game.getGrid()[i];
58              StackPane spane = tiles[tileNumber-1];
59              spane.setTranslateX((i%4)*size);
60              spane.setTranslateY((i/4)*size);
61          }
62          
63      }
64      
65      public void solve() {
66          
67          AnimationTimer timer = new AnimationTimer() {
68              private int i = 0;
69              private long lastUpdate = 0; 
70              private GameOfFifteen helpGame = new GameOfFifteen();
71              private GameSolver solver = new GameSolver(game);
72              private int[] moves;
73              
74              @Override
75              public void start() {
76                  helpGame.setGrid(game.getGrid().clone());
77                  moves = solver.solver();
78                  if (moves.length == 1 && moves[0] == 0) {
79                      this.stop();
80                      return;
81                  }
82                  super.start();
83              }
84              @Override
85              public void stop() {
86                  shuffle.setDisable(false);
87                  super.stop();
88              }
89              @Override
90              public void handle(long now) {
91                  if (now-lastUpdate >= 1000000*duration) {
92  
93                      int blank = helpGame.getBlank();
94                      int tileNumber = moves[i];
95                      StackPane spane = tiles[tileNumber-1];
96                      double tileIndex = (spane.getTranslateX()/size) + (spane.getTranslateY()/size)*4;
97                      int tileI = (int) tileIndex;
98              
99                      if (tileI == blank-4) {
100                         moveDown(spane);
101                         helpGame.goUp();
102                         
103                     }
104                     else if (tileI == blank-1) {
105                         //moveright here and logic left
106                         moveRight(spane);
107                         helpGame.goLeft();
108                     }
109                     else if (tileI == blank+1) {
110                         //left and right
111                         moveLeft(spane);
112                         helpGame.goRight();
113                     }
114                     else if (tileI == blank+4) {
115                         moveUp(spane);
116                         helpGame.goDown();
117                     }
118                     lastUpdate = now;
119                     i++;
120                     if (i == moves.length) {
121                         this.stop();
122                     }
123                 }
124                 
125             }
126         };
127         timer.start();
128     }

130     public Pane createScene() {
131         root.setPrefSize(575, 400);
132         BackgroundFill bg = new BackgroundFill(Color.GREY, new CornerRadii(0), new Insets(0, 175, 0, 0));
133         root.setBackground(new Background(bg));
134 
135         Button solve = new Button("Solve");
136         solve.setShape(new Circle(60));
137         solve.setMinSize(60, 60);
138         String styles = "-fx-background-color: red;"
139                 + "-fx-border-color: black;"
140                 + "-fx-text-fill: gold;";
141         solve.setStyle(styles);
142         solve.setTranslateX(450);
143         solve.setTranslateY(150);
144         shuffle = new Button("Shuffle");
145         shuffle.setShape(new Circle(60));
146         shuffle.setMinSize(60, 60);
147         shuffle.setTranslateX(450);
148         shuffle.setTranslateY(50);
149         shuffle.setStyle(styles); 
150         
151         
152         solve.setOnMouseClicked((event) -> {
153             shuffle.setDisable(true);
154             solve();
155         });
156         
157         shuffle.setOnMouseClicked((event) -> {
158             shuffle();
159         });
160         
161         
162         StackPane s1 = new StackPane();
163         StackPane s2 = new StackPane();
164         StackPane s3 = new StackPane();
165         StackPane s4 = new StackPane();
166         StackPane s5 = new StackPane();
167         StackPane s6 = new StackPane();
168         StackPane s7 = new StackPane();
169         StackPane s8 = new StackPane();
170         StackPane s9 = new StackPane();
171         StackPane s10 = new StackPane();
172         StackPane s11 = new StackPane();
173         StackPane s12 = new StackPane();
174         StackPane s13 = new StackPane();
175         StackPane s14 = new StackPane();
176         StackPane s15 = new StackPane();
177         
178         tiles = new StackPane[]{s1, s2, s3, s4, s5, s6, s7, s8, s9,
179             s10, s11, s12, s13, s14, s15};        
180         
181         for (int i = 0; i < 15; i++) {
182             int x = (i % 4) * size;
183             int y = (i / 4) * size;
184             tiles[i].setPrefSize(size, size);
185             tiles[i].setTranslateX(x);
186             tiles[i].setTranslateY(y);
187             
188             Text text = createText(i);
189             Rectangle rect = createRectangle(i);
190             tiles[i].getChildren().addAll(rect, text);
191             root.getChildren().addAll(tiles[i]);
192         }
193 
194         Line line = new Line();
195         line.setFill(Color.BLACK);
196         line.setStartX(4*size +1);
197         line.setStartY(0);
198         line.setEndX(4*size +1);
199         line.setEndY(4*size);
200         line.setStrokeWidth(2);
201         
202         root.getChildren().addAll(line, solve, shuffle);
203         
204         return root;
205     }
206     public Text createText(int i) {
207         Text text = new Text(Integer.toString(i+1));
208         text.setFill(Color.GOLD);
209         text.setFont(new Font(fontSize));
210         return text;
211     }
212     public Rectangle createRectangle(int i) {
213         Rectangle rect = new Rectangle();
214         if (i % 2 == 0) {
215             rect.setFill(Color.WHITE);
216         }
217         else {
218             rect.setFill(Color.RED);
219         }
220         rect.setArcHeight(10);
221         rect.setArcWidth(10);
222         rect.setHeight(size);
223         rect.setWidth(size);
224         rect.setStroke(Color.BLACK);
225         return rect;
226     }

229     public void moveLeft(Node node) {
230         TranslateTransition left = new TranslateTransition();
231         left.setDuration(Duration.millis(duration));
232         double from = node.getTranslateX();
233         left.setFromX(from);
234         left.setToX(from-size);
235         left.setNode(node);
236         left.play();
237     }
238     public void moveRight(Node node) {
239         TranslateTransition right = new TranslateTransition();
240         right.setDuration(Duration.millis(duration));
241         double from = node.getTranslateX();
242         right.setFromX(from);
243         right.setToX(from+size);
244         right.setNode(node);
245         right.play();
246     }
247     public void moveDown(Node node) {
248         TranslateTransition down = new TranslateTransition();
249         down.setDuration(Duration.millis(duration));
250         double from = node.getTranslateY();
251         down.setFromY(from);
252         down.setToY(from+size);
253         down.setNode(node);
254         down.play();
255     }
256     public void moveUp(Node node) {
257         TranslateTransition up = new TranslateTransition();
258         up.setDuration(Duration.millis(duration));
259         double from = node.getTranslateY();
260         up.setFromY(from);
261         up.setToY(from-size);

263         up.play();
264     }
