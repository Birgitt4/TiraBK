4   /**
5    * My implementation of a priorityQueue.
6    * Min value always on peek of the queue, in this case min fScore
7    */
8   public class PriorityQueue {
9       private Node[] pqueue;
10      private int next;
11      
12      /**
13       * constructor, creates new priorityQueue.
14       */
15      public PriorityQueue() {
16          pqueue = new Node[5];
17          next = 1;
18      }
19      public Node[] getArray() {
20          return pqueue;
21      }
22      
23      /**
24       * add the given node to this queue and puts the queue in right order after.
25       * @param node Node we want to be added in this queue
26       */
27      public void add(Node node) {
28          int i = next;
29          if (next == pqueue.length) {
30              increaseSize();
31          }
32          pqueue[next] = node;
33          next++;
34          
35          while (i > 1 && pqueue[i / 2].getFScore() > pqueue[i].getFScore()) {
36              Node temp = pqueue[i / 2];
37              pqueue[i / 2] = pqueue[i];
38              pqueue[i] = temp;
39              i = i / 2;
40          }
41      }
42      
43      /**
44       * Retrieves and removes first node in this queue.
45       * @return node with min f score
46       */
47      public Node poll() {
48          Node node = pqueue[1];
49          next--;
50          int i = 1;
51          pqueue[1] = pqueue[next];
52          pqueue[next] = null;
53          Node temp = pqueue[i];
54          while (hasBothKids(i)) {
55              int j = smaller(i);
56              pqueue[i] = pqueue[j];
57              pqueue[j] = temp;
58              i = j;
59          }
60          if (hasLeftKid(i)) {
61              if (pqueue[2 * i].getFScore() < pqueue[i].getFScore()) {
62                  pqueue[i] = pqueue[2 * i];
63                  pqueue[2 * i] = temp;
64              }
65          }
66          return node;
67      }
68      private boolean hasBothKids(int i) {
69          if (2 * i + 1 < pqueue.length) {
70              if (pqueue[2 * i + 1] != null) {
71                  return true;
72              }
73          }
74          return false;
75      }
76      private boolean hasLeftKid(int i) {
77          if (2 * i < pqueue.length) {
78              if (pqueue[2 * i] != null) {
79                  return true;
80              }
81          } 
82          return false;
83      }
84      private int smaller(int i) {
85          if (pqueue[2 * i].getFScore() < pqueue[2 * i + 1].getFScore()) {
86              return 2 * i;
87          } else {
88              return 2 * i + 1;
89          }
90      }
91      private void increaseSize() {
92          Node[] temp = new Node[2 * next];
93          for (int i = 0; i < next; i++) {
94              temp[i] = pqueue[i];
95          }
96          pqueue = temp;
97      }
98      
99      public boolean isEmpty() {
100         return pqueue[1] == null;
101     }
102 }
