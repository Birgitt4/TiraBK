4   /**
5    * Node is a way to save important information of one state of the game.
6    * 
7    */
8   public class Node {
9       
10      private int[] route;
11      private int[] grid;
12      private int manhattan;
13      private int linearrow;
14      private int linearcol;
15      private int gScore;
16      private char direction;
17      
18      /**
19       * constructor.
20       * @param route route that tells what numbers have been moved since start to
21       * this state
22       * @param gamestate order of the tiles when this node is created
23       * @param manhattan grids sum of manhattan distances
24       * @param linearrow linearconflicts that occurs in rows
25       * @param linearcol linearconflicts that occurs in columns
26       * @param gScore how many moves have been taken
27       * @param direction the direction of the blanks last move
28       */
29      public Node(int[] route, int[] gamestate, int manhattan, int linearrow, int linearcol, int gScore, char direction) {
30          this.route = route;
31          grid = gamestate;
32          this.manhattan = manhattan;
33          this.linearrow = linearrow;
34          this.linearcol = linearcol;
35          this.gScore = gScore;
36          this.direction = direction;
37      }
38      public int getGScore() {
39          return gScore;
40      }
41      public int getManhattan() {
42          return manhattan;
43      }
44      public void setManhattan(int manhattan) {
45          this.manhattan = manhattan;
46      }
47      public int getLinearrow() {
48          return linearrow;
49      }
50      public void setLinearrow(int linearrow) {
51          this.linearrow = linearrow;
52      }
53      public int getLinearcol() {
54          return linearrow;
55      }
56      public void setLinearcol(int linearcol) {
57          this.linearcol = linearcol;
58      }
59      public int getFScore() {
60          return gScore + manhattan + linearrow + linearcol;
61      }
62      public void setGScore(int g) {
63          this.gScore = g;
64      }
65      public int[] getRoute() {
66          return route;
67      }
68      public int[] getGrid() {
69          return grid;
70      }
71      public char getDirection() {
72          return direction;
73      }
74  }
